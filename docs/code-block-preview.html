2│ if len(arr) <= 1:
3│ return arr
4│ pivot = arr[len(arr) // 2]
5│ left = [x for x in arr if x < pivot]
6│ mid = [x for x in arr if x == pivot]
7│ right = [x for x in arr if x > pivot]
8│ return quicksort(left) + mid + quicksort(right)
